    const DIFFS  = ['全部','基礎','標準','発展'];

    const LENGTHS = [5, 8, 10, 15];

 

    const storeKey = 'rika_quiz_v1';

    const statsKey = 'rika_quiz_stats_v1';

 

    const defaultState = {

      topic: '全部',

      diff: '全部',

      length: 8,

      seed: Math.random().toString(36).slice(2),

    };

 

    const defaultStats = {

      ok: 0,

      ng: 0,

      streak: 0,

      best: 0,

    };

 

    let state = loadJSON(storeKey, defaultState);

    let stats = loadJSON(statsKey, defaultStats);

 

    let pool = [];

    let idx = 0;

    let locked = false;

    let answeredThisQuestion = false;

 

    // ---------- DOM ----------

    const el = (id) => document.getElementById(id);

 

    const accEl = el('acc');

    const topicEl = el('topic');

    const diffEl = el('diff');

    const counterEl = el('counter');

    const progEl = el('prog');

    const qEl = el('question');

    const choicesEl = el('choices');

    const explainBox = el('explain');

    const explainText = el('explainText');

 

    const nextBtn = el('nextBtn');

    const shuffleBtn = el('shuffleBtn');

    const resetBtn = el('resetBtn');

    const exportBtn = el('exportBtn');

 

    const okEl = el('ok');

    const ngEl = el('ng');

    const streakEl = el('streak');

    const bestEl = el('best');

 

    const toastEl = el('toast');

 

    // ---------- UI: chips ----------

    const topicChips = el('topicChips');

    const diffChips  = el('diffChips');

    const lenChips   = el('lenChips');

 

    function mkChip(label, active, onClick){

      const d = document.createElement('div');

      d.className = 'chip' + (active ? ' active' : '');

      d.textContent = label;

      d.addEventListener('click', onClick);

      return d;

    }

 

    function renderChips(){

      topicChips.innerHTML = '';

      diffChips.innerHTML = '';

      lenChips.innerHTML = '';

 

      TOPICS.forEach(t => {

        topicChips.appendChild(mkChip(t, state.topic === t, () => {

          state.topic = t;

          saveState();

          rebuild();

        }));

      });

 

      DIFFS.forEach(d => {

        diffChips.appendChild(mkChip(d, state.diff === d, () => {

          state.diff = d;

          saveState();

          rebuild();

        }));

      });

 

      LENGTHS.forEach(n => {

        lenChips.appendChild(mkChip(String(n), state.length === n, () => {

          state.length = n;

          saveState();

          rebuild();

        }));

      });

    }

 

    // ---------- プール作成 ----------

    function buildPool(){

      let list = QUESTION_BANK.slice();

 

      if(state.topic !== '全部') list = list.filter(x => x.topic === state.topic);

      if(state.diff !== '全部')  list = list.filter(x => x.diff === state.diff);

 

      // シャッフル

      list = shuffle(list);

 

      // lengthに合わせる

      list = list.slice(0, Math.min(state.length, list.length));

 

      // もし0なら全部から少しだけ

      if(list.length === 0){

        list = shuffle(QUESTION_BANK.slice()).slice(0, Math.min(state.length, QUESTION_BANK.length));

      }

 

      return list;

    }

 

    function rebuild(){

      pool = buildPool();

      idx = 0;

      locked = false;

      answeredThisQuestion = false;

      nextBtn.disabled = true;

      explainBox.style.display = 'none';

      renderChips();

      renderStats();

      renderQuestion();

      toast('設定を反映したよ');

    }

 

    // ---------- 問題表示 ----------

    function renderQuestion(){

      const total = pool.length;

      const q = pool[idx];

 

      counterEl.textContent = `${Math.min(idx+1,total)} / ${total}`;

      progEl.style.width = total ? `${(idx/total)*100}%` : '0%';

 

      if(!q){

        qEl.textContent = '問題がありません。設定を変えてみてね。';

        choicesEl.innerHTML = '';

        topicEl.textContent = '—';

        diffEl.textContent = '—';

        nextBtn.disabled = true;

        return;

      }

 

      topicEl.textContent = q.topic;

      diffEl.textContent = q.diff;

      qEl.textContent = q.q;

 

      explainBox.style.display = 'none';

      explainText.textContent = '';

 

      choicesEl.innerHTML = '';

      q.choices.forEach((c, i) => {

        const b = document.createElement('button');

        b.className = 'btn';

        b.type = 'button';

        b.dataset.i = String(i);

        b.innerHTML = `<div class="tag">${i+1}</div><div>${escapeHTML(c)}</div>`;

        b.addEventListener('click', () => pick(i));

        choicesEl.appendChild(b);

      });

 

      locked = false;

      answeredThisQuestion = false;

      nextBtn.disabled = true;

 

      // 進捗：今の問題番号まで含めて伸ばす

      progEl.style.width = total ? `${(idx/total)*100}%` : '0%';

    }

 

    function pick(i){

      if(locked) return;

      locked = true;

      answeredThisQuestion = true;

 

      const q = pool[idx];

      const isOK = (i === q.answer);

 

      // ボタンの見た目

      const btns = [...choicesEl.querySelectorAll('button')];

      btns.forEach(b => {

        b.disabled = true;

        const bi = Number(b.dataset.i);

        if(bi === q.answer) b.classList.add('good');

        else if(bi === i) b.classList.add('bad');

      });

 

      // 解説

      explainText.textContent = q.exp;

      explainBox.style.display = 'block';

 

      // 成績

      if(isOK){

        stats.ok += 1;

        stats.streak += 1;

        stats.best = Math.max(stats.best, stats.streak);

        toast('正解！');

      }else{

        stats.ng += 1;

        stats.streak = 0;

        toast('惜しい！');

      }

 

      saveStats();

      renderStats();

 

      nextBtn.disabled = false;

      progEl.style.width = pool.length ? `${((idx+1)/pool.length)*100}%` : '0%';

    }

 

    function next(){

      if(!answeredThisQuestion) return;

      idx += 1;

 

      if(idx >= pool.length){

        // 終了

        toast('このセットは終了！シャッフルで次へ');

        qEl.textContent = 'セット終了！🎉\n「問題をシャッフル」で次のセットを始めよう。';

        choicesEl.innerHTML = '';

        explainBox.style.display = 'none';

        nextBtn.disabled = true;

        topicEl.textContent = '—';

        diffEl.textContent = '—';

        counterEl.textContent = `${pool.length} / ${pool.length}`;

        progEl.style.width = '100%';

        return;

      }

 

      renderQuestion();

    }

 

    // ---------- 成績 ----------

    function renderStats(){

      okEl.textContent = stats.ok;

      ngEl.textContent = stats.ng;

      streakEl.textContent = stats.streak;

      bestEl.textContent = stats.best;

 

      const total = stats.ok + stats.ng;

      const acc = total ? Math.round((stats.ok/total)*100) : 0;

      accEl.textContent = `${acc}%`;

    }

 

    function resetStats(){

      if(!confirm('成績をリセットする？（この端末の記録が消えます）')) return;

      stats = structuredClone(defaultStats);

      saveStats();

      renderStats();

      toast('リセットしたよ');

    }

 

    // ---------- 書き出し ----------

    function exportBank(){

      const data = JSON.stringify(QUESTION_BANK, null, 2);

      // クリップボード

      navigator.clipboard?.writeText(data).then(() => {

        toast('問題データをコピーしたよ');

      }).catch(() => {

        // フォールバック

        const w = window.open();

        w.document.write('<pre style="white-space:pre-wrap;word-break:break-word;">'+escapeHTML(data)+'</pre>');

        toast('別タブに表示したよ');

      });

    }

 

    // ---------- 保存 ----------

    function saveState(){

      localStorage.setItem(storeKey, JSON.stringify(state));

    }

    function saveStats(){

      localStorage.setItem(statsKey, JSON.stringify(stats));

    }

    function loadJSON(key, fallback){

      try{

        const raw = localStorage.getItem(key);

        if(!raw) return structuredClone(fallback);

        const obj = JSON.parse(raw);

        return { ...structuredClone(fallback), ...obj };

      }catch{

        return structuredClone(fallback);

      }

    }

 

    // ---------- ユーティリティ ----------

    function shuffle(arr){

      const a = arr.slice();

      for(let i=a.length-1;i>0;i--){

        const j = Math.floor(Math.random()*(i+1));

        [a[i], a[j]] = [a[j], a[i]];

      }

      return a;

    }

 

    function toast(msg){

      toastEl.textContent = msg;

      toastEl.classList.add('show');

      clearTimeout(toast._t);

      toast._t = setTimeout(() => toastEl.classList.remove('show'), 1100);

    }

 

    function escapeHTML(s){

      return String(s)

        .replaceAll('&','&amp;')

        .replaceAll('<','&lt;')

        .replaceAll('>','&gt;')

        .replaceAll('"','&quot;')

        .replaceAll("'",'&#039;');

    }

 

    // ---------- イベント ----------

    nextBtn.addEventListener('click', next);

    shuffleBtn.addEventListener('click', () => {

      state.seed = Math.random().toString(36).slice(2);

      saveState();

      rebuild();

    });

    resetBtn.addEventListener('click', resetStats);

    exportBtn.addEventListener('click', exportBank);

 

    // キーボード操作（PC用）

    window.addEventListener('keydown', (e) => {

      if(e.key >= '1' && e.key <= '4'){

        const i = Number(e.key) - 1;

        const b = choicesEl.querySelector(`button[data-i="${i}"]`);

        if(b && !b.disabled) b.click();

      }

      if(e.key === 'Enter'){

        if(!nextBtn.disabled) nextBtn.click();

      }

    });

 

    // ---------- 起動 ----------

    renderChips();

    renderStats();

    pool = buildPool();

    renderQuestion();

  </script>

</body>

</html>
